Lab-18

Part-A

1) SELECT CONCAT(EName, ' works at ', City, ' and earns ', Salary) FROM Employee;

2) SELECT COUNT(*) FROM Employee WHERE Salary > 5000;

3) SELECT CONCAT(SUBSTRING(EName, 1, 4), SUBSTRING(EName, -3)) AS EName FROM Employee;

4) SELECT City FROM Employee WHERE JoiningDate > '2014-01-01' GROUP BY City HAVING SUM(Salary) > 15000;

5) SELECT REPLACE(EName, 'u', 'oo') FROM Employee;


Part-B

1) SELECT City, AVG(Salary) AS AvgSalary, COUNT(*) AS TotalEmployees FROM Employee GROUP BY City;

2) SELECT SUM(Salary) AS TotalSalary FROM Employee WHERE Gender = 'Female';

3) SELECT EName, DATEDIFF(YEAR, JoiningDate, GETDATE()) AS Experience FROM Employee;

4) ALTER TABLE Employee DROP COLUMN Department;

5) UPDATE Employee SET City = NULL WHERE City = 'Sydney';


Part-C

1) SELECT EName, Salary, JoiningDate FROM Employee;

2) SELECT LENGTH(EName) + LENGTH(Gender) AS CombinedLength FROM Employee;

3) SELECT MAX(Salary) - MIN(Salary) AS SalaryDifference FROM Employee;

4) ALTER TABLE Employee RENAME COLUMN EName TO FirstName;

5) ALTER TABLE Employee RENAME TO EmpMaster;
