CREATE TABLE Student (
    Rno INT PRIMARY KEY,
    Name VARCHAR(255) NOT NULL,
    City VARCHAR(255) NOT NULL,
    DID INT NOT NULL
);

CREATE TABLE Academic (
    Rno INT PRIMARY KEY,
    SPI FLOAT CHECK (SPI >= 0),
    Bklog INT CHECK (Bklog >= 0)
);

CREATE TABLE Department (
    DID INT PRIMARY KEY,
    DName VARCHAR(255) NOT NULL
);

Part-A

1) SELECT * FROM Student WHERE DID = (SELECT DID FROM Department WHERE DName = 'Computer');

2) SELECT Name FROM Student WHERE Rno IN (SELECT Rno FROM Academic WHERE SPI > 8);

4) SELECT COUNT(*) FROM Student WHERE DID = (SELECT DID FROM Department WHERE DName = 'Electrical');
